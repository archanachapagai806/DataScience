# Initialize variables to store the salary sum, number of employees, skewness sum, and squared differences sum
salary_sum = 0
num_employees = 0
skew_sum = 0
squared_diff_sum = 0

# Calculate the sum of all the salaries in the column
for salary in df["salary_in_usd"]:
    salary_sum += salary
    num_employees += 1

# Calculate the mean of the salaries
mean_salary = salary_sum / num_employees

# Calculate the squared differences from the mean for standard deviation
for salary in df["salary_in_usd"]:
    squared_diff_sum += (salary - mean_salary) ** 2

# Calculate the standard deviation
salary_std = (squared_diff_sum / (num_employees - 1)) ** 0.5

# Calculate the cubed differences from the mean for skewness
for salary in df["salary_in_usd"]:
    skew_sum += (salary - mean_salary) ** 3

# Calculate the skewness
skew_numerator = skew_sum / num_employees
salary_skewness = skew_numerator / (salary_std ** 3)

# Calculate mean, standard deviation, skewness, and kurtosis of a column in a pandas DataFrame

salary_sum = 0
num_employees = 0
squared_diff_sum = 0
kurtosis_sum = 0

# Calculate the sum of all the salaries in the column
for salary in df["salary_in_usd"]:
    salary_sum += salary
    num_employees += 1

# Calculate the mean of the salaries
mean_salary = salary_sum / num_employees

# Calculate the squared differences from the mean for standard deviation
for salary in df["salary_in_usd"]:
    squared_diff_sum += (salary - mean_salary) ** 2

# Calculate the standard deviation
salary_std = (squared_diff_sum / (num_employees - 1)) ** 0.5

# Calculate the fourth power differences from the mean for kurtosis
for salary in df["salary_in_usd"]:
    kurtosis_sum += (salary - mean_salary) ** 4

kurtosis_numerator = kurtosis_sum / num_employees
salary_kurtosis = (kurtosis_numerator / (salary_std ** 4)) - 3

# Print the kurtosis of the salary_in_usd column
print(f"Kurtosis of salary_in_usd: {salary_kurtosis}")
